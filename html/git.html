git config --global user.name "dabin"

git config --global user.email lwb@jd.com

git config --list

git init

git add *.html //存储到暂存区

git commit -m 'submit' //提交到分支

git status

git clone [url]

git reset HEAD //暂存区被master取代

git rm --cached x.html //从暂存区删除此文件

git checkout [x.html] //用暂存区的文件替换工作区文件！

git checkout HEAD [X.html] //用master取代暂存区和工作区的文件！

git branch //查看分支

git branch branchname //创建分支

git checkout branchname //切换分支

git branch -d branchname //删除分支

git merge branchname //将此分支合并到主分支

git diff //查看冲突

git log [--oneline,--reverse,--author] //查看提交历史

函数重载:arguments

function makeArray(){
 var arr = [];
 for(var i=0; i<arguments.length; i++){
  arr.push(arguments[i]);
 }
 return arr;
}

类型检查:typeof,constructor

function displayError(msg){
 if(typeof msg == 'undefined'){
  alert('参数居然为空！');
 }else{
  alert(msg);
 }
 if(typeof num == 'string'){
  num = parseInt(num);
 }
 if(str.constructor == Array){
  str = str.join(',');
 }
}
